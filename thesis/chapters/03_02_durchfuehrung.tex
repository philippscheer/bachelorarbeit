\subsection{Extraktionsprozess}\label{0302_daten_extraktion}\label{chap:scraping}


Der Prozess der automatisierten Extraktion wurde folgendermaßen realisiert:


\begin{algorithm}[H]
  \caption{Datenextraktion aus Vorlesungsverzeichnis}
  \label{alg:datenextraktion}
  \begin{algorithmic}[1]
    \State $\mathit{offerings} \gets \emptyset$
    \State $\mathit{courses} \gets \emptyset$

    \For{$id \gets 1, 9999$}
    \State $url \gets \textsc{CraftCourseURL}(id)$
    \State $page \gets \textsc{HttpGet}(url)$

    \If{$\textsc{Contains}(page\, "Keine~Lehrveranstaltung~gefunden")$}
    \State \textbf{continue}
    \EndIf

    \State $(course, dates) \gets \textsc{ExtractDates}(page)$

    \If{$course \notin \text{dom}(courses)$}
    \State $ects \gets \textsc{ExtractECTSFromCourse}(course)$
    \State $courses \gets courses \cup \{(course, ects)\}$
    \EndIf

    \State $\mathit{offerings} \gets \mathit{offerings} \cup \{(id, dates, courses_{course})\}$
    \EndFor

  \end{algorithmic}
\end{algorithm}
